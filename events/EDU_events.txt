#Prussian revolt
country_event = {
	id = EDU.1
	title = EDU.1.t
	desc = EDU.1.d
	#picture = 
	trigger = {
		tag = EDU
		has_idea = EDU_ethnic_tensions
		has_stability < 0.3
	}

	fire_only_once = yes

	option = {
		name = EDU.1.a
		start_civil_war = {
			ideology = fascism
			size = 0.4
			states = { 68 61 67 64 66 62 63 }
		}
		random_country = {
			limit = {
				original_tag = EDU
				has_government = fascism
			}
			save_event_target_as = EDU_german_revolt
		}
		DDR = {
			annex_country = { target = event_target:EDU_german_revolt transfer_troops = yes }
		}
		declare_war_on = {
			target = DDR 
			type = annex_everything
		}
	}
}
