#Регион восстает против ВСИ
country_event = {
	id = FRU.4
	title = FRU.4.t
	desc = FRU.4.d
	#picture = GFX_report_event_chinese_soldiers

	trigger = {
		original_tag = UKR
		has_government = fascism
		FRU = {
			exists = yes
			has_government = communism
		}
		communism > 0.3
	}

	mean_time_to_happen = {
		days = 270
		modifier = {
			factor = 0.5
			communism > 0.5
		}
	}

	immediate = {
		hidden_effect = {
			random_owned_controlled_state = {
				limit = {
					any_neighbor_state = {
						owner = { tag = FRU }
					}
				}
				save_event_target_as = UKR_communist_state
			}
		}
	}

	option = {
		name = FRU.4.a
		PRC = {
			transfer_state = event_target:UKR_communist_state
			country_event = { id = FRU.5 days = 1 }
		}
		add_popularity = {
			ideology = communism
			popularity = -0.15
		}
	}
}
