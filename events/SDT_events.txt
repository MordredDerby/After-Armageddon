country_event = {
	id = SDT.4
	title = SDT.4.t
	desc = SDT.4.d

	fire_only_once = yes

	mean_time_to_happen = {
		days = 5
	}

	trigger = {
		tag = AUS
		AND = {
			check_variable = { sdt_germany > sdt_austria }
			check_variable = { sdt_germany > sdt_czechia }
		}
	}
	#Accept terms
	option = {
		name = SDT.4.a
		add_political_power = -50
		release = SDT
		GER = {
			set_autonomy = {
				target = SDT
				autonomous_state = autonomy_integrated_puppet
			}
		}
	}
	#Refuse
	option = {
		name = SDT.4.b
		add_stability = -0.1
	}
}
