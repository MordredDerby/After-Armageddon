add_namespace = SIB

news_event = {
	id = SIB.1
	title = SIB.1.t
	desc = SIB.1.d
	picture = GFX_news_event_kuznetsk

	trigger = {
		tag = SIB
		date > 1961.6.6
	}

	major = yes
	fire_only_once = yes

	option = {
		name = SIB.1.a
		set_global_flag	= SIB_decline
		SIB = {
			set_temp_variable = { var = siberia_defenders value = 0 }
			set_temp_variable = { var = siberia_attackers value = 0 }
		}
	}
}

country_event = {
	id = SIB.2
	title = SIB.2.t
	desc = SIB.2.d

	fire_only_once = yes

	trigger = {
		tag = SIB
		URA = { 
			has_country_flag = decided_siberia
		}
		TRK = { 
			has_country_flag = decided_siberia
		}
		MON = { 
			has_country_flag = decided_siberia
		}
		MAN = { 
			has_country_flag = decided_siberia
		}
	}

	#Give up
	option = {
		name = SIB.2.a
		set_country_flag = SIB_surrendered
		if = {
			limit = {
				has_global_flag = TRK_partition_siberia
			}
			TRK = {
				transfer_state = 590
				transfer_state = 588
			}
		}
		if = {
			limit = {
				has_global_flag = MAN_partition_siberia
			}
			MAN = {
				transfer_state = 561
				transfer_state = 562
				transfer_state = 637
				transfer_state = 1090
			}
		}
		if = {
			limit = {
				has_global_flag = URA_partition_siberia
			}
			URA = {
				transfer_state = 403
				transfer_state = 579
				transfer_state = 580
			}
		}
		if = {
			limit = {
				has_global_flag = MON_partition_siberia
			}
			release = YAK
			MON = {
				transfer_state = 654
				puppet = YAK
			}
			BUR = {
				transfer_state = 565
				transfer_state = 566
			}
		}
	}
	#Rely on defender
	option = {
		name = SIB.2.b
		trigger = { 
			check_variable = {
				var = siberia_defenders
				value = 0
				compare = more_than
			}
		}
		if = {
			limit = {
				check_variable = {
					var = siberia_defenders
					value = 1
					compare = equal
				}
			}
			random_country = {
				limit = {
					has_country_flag = siberia_defender
				}
				puppet = SIB
			}
		}
		every_country = {
			limit = {
				has_country_flag = siberia_attacker
			}
			country_event =  { id = SIB.3 }
		}
	}
	#Resist
	option = {
		name = SIB.2.c
		trigger = { 
			check_variable = {
				var = siberia_defenders
				value = 1
				compare = less_than
			}
		}
		every_country = {
			limit = {
				has_country_flag = siberia_attacker
			}
			country_event =  { id = SIB.3 }
		}
	}
}

	desc = SIB.5.d
	picture = GFX_report_event_soviet_tank_parade

	fire_only_once = yes
	mean_time_to_happen = { days = 21 }
	trigger = {
		tag = SIB
		OR = {
			NOT = { 
				has_country_flag = SIB_moved_capital
				controls_state = 569
			}
			has_country_flag = SIB_surrendered
		}
	}

	option = {
		name = SIB.5.a
		set_politics = {
			ruling_party = centrist
		}
		add_popularity = {
    		ideology = radical_rightist
    		popularity = -0.3
		}
		add_popularity = {
    		ideology = centrist
    		popularity = 0.1
		}
		random_unit_leader = {
			limit = {
				has_id = 331922
			}
			promote_leader = yes
			add_unit_leader_trait = politically_connected
		}
	}
}
